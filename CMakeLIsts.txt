cmake_minimum_required(VERSION 3.15)
project(hotspot C)

set(CMAKE_C_STANDARD 99)

# Build options
option(USE_SUPERLU "Build with SuperLU acceleration" OFF)
option(ENABLE_DEBUG "Enable debug build" OFF)
option(ENABLE_PROFILE "Enable profiling (-pg)" OFF)
set(VERBOSE_LEVEL "1" CACHE STRING "Verbosity level (0-3)")
set(MATHACCEL "0" CACHE STRING "Use math acceleration (e.g., MKL, OpenBLAS)")

# Compiler flags
if(ENABLE_DEBUG)
  add_compile_options(-O0 -ggdb -Wall -Wextra)
elseif(ENABLE_PROFILE)
  add_compile_options(-O3 -pg -ggdb -Wall -Wextra)
else()
  add_compile_options(-O3)
endif()

include(CheckCCompilerFlag)
check_c_compiler_flag("-mavx2" COMPILER_SUPPORTS_AVX2)

if(COMPILER_SUPPORTS_AVX2)
  add_compile_options(-mavx2)
endif()

# Find required libraries
include(CMakeFindDependencyMacro)

# Math library (always required)
if(NOT WIN32)
  find_library(MATH_LIBRARY m)

  if(NOT MATH_LIBRARY)
    message(FATAL_ERROR "Math library (libm) not found")
  endif()
else()
  # Windows 下数学库为空
  set(MATH_LIBRARY "")
endif()

set(MKL_FOUND OFF)

if(MATHACCEL EQUAL "1")
  find_package(MKL REQUIRED)

  if(MKL_FOUND)
    message(STATUS "Intel MKL found automatically!")

    if(DEFINED MKL_INCLUDE_DIRS)
      include_directories(${MKL_INCLUDE_DIRS})
    elseif(DEFINED MKL_INCLUDE)
      include_directories(${MKL_INCLUDE})
    endif()
  else()
    message(FATAL_ERROR "Intel MKL not found. Please install Intel oneAPI MKL or set MATHACCEL=OFF")
  endif()
endif()

# SuperLU dependencies (optional)
if(USE_SUPERLU)
  set(CPM_DOWNLOAD_VERSION 0.40.2)
  set(CPM_DOWNLOAD_URL https://github.com/cpm-cmake/CPM.cmake/releases/download/v${CPM_DOWNLOAD_VERSION}/CPM.cmake)

  if(NOT EXISTS ${CMAKE_BINARY_DIR}/cmake/CPM.cmake)
    file(DOWNLOAD ${CPM_DOWNLOAD_URL} ${CMAKE_BINARY_DIR}/cmake/CPM.cmake)
  endif()

  include(${CMAKE_BINARY_DIR}/cmake/CPM.cmake)

  CPMAddPackage(
    NAME SuperLU
    GITHUB_REPOSITORY xiaoyeli/superlu
    VERSION 6.0.0
    OPTIONS
    "BUILD_SHARED_LIBS OFF" # 编译成静态库（最稳，无DLL）
    "CMAKE_BUILD_TYPE Release"
    "enable_tests OFF" # 关闭测试，加快编译
    "CMAKE_POLICY_VERSION_MINIMUM 3.5"
  )
endif()

# Source files
set(COMMON_SOURCES
  src/microchannel.c src/materials.c src/flp.c src/flp_desc.c
  src/npe.c src/shape.c src/temperature.c src/RCutil.c
  src/package.c src/temperature_block.c src/temperature_grid.c
  src/util.c src/wire.c
)

# Common library
add_library(hotspot_common STATIC ${COMMON_SOURCES})
target_include_directories(hotspot_common PUBLIC
  ${CMAKE_SOURCE_DIR}/src
)
target_compile_definitions(hotspot_common PRIVATE
  VERBOSE=${VERBOSE_LEVEL}
  SUPERLU=$<BOOL:${USE_SUPERLU}>
  MATHACCEL=${MATHACCEL}
)

# Link libraries helper
set(LINK_LIBS ${MATH_LIBRARY})

# When math acceleration is requested, prefer MKL (if found) otherwise fall
# back to the generic BLAS/LAPACK variables. Append to LINK_LIBS rather
# than overwriting it (previous ordering accidentally discarded appended libs).
if(MATHACCEL EQUAL "1")
  if(TARGET MKL::MKL)
    list(APPEND LINK_LIBS MKL::MKL)
  elseif(DEFINED MKL_LIBRARIES)
    list(APPEND LINK_LIBS ${MKL_LIBRARIES})
  elseif(DEFINED MKL_LIBRARY)
    list(APPEND LINK_LIBS ${MKL_LIBRARY})
  else()
    list(APPEND LINK_LIBS ${BLAS_LIBRARIES} ${LAPACK_LIBRARIES})
  endif()
endif()

if(USE_SUPERLU)
  target_link_libraries(hotspot_common PUBLIC superlu)
endif()

# Executables
add_executable(hotspot src/hotspot.c)
target_link_libraries(hotspot PRIVATE hotspot_common ${LINK_LIBS})

add_executable(hotfloorplan src/hotfloorplan.c)
target_link_libraries(hotfloorplan PRIVATE hotspot_common ${LINK_LIBS})

# Static library
add_library(libhotspot STATIC ${COMMON_SOURCES})
set_target_properties(libhotspot PROPERTIES OUTPUT_NAME hotspot)

# Copy the executables from build to the project root for easier access
add_custom_command(TARGET hotspot POST_BUILD
  COMMAND ${CMAKE_COMMAND} -E copy $<TARGET_FILE:hotspot> ${CMAKE_SOURCE_DIR}/hotspot
)
add_custom_command(TARGET hotfloorplan POST_BUILD
  COMMAND ${CMAKE_COMMAND} -E copy $<TARGET_FILE:hotfloorplan> ${CMAKE_SOURCE_DIR}/hotfloorplan
)
